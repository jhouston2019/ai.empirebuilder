// Credit Management System class CreditSystem { constructor() { this.creditCosts = { textGeneration: 1, // per 1000 tokens imageGeneration: 10, // per image audioTranscription: 5, // per minute advancedAnalysis: 20 // per document }; } calculateCreditsNeeded(operation, units) { return this.creditCosts[operation] * units; } async processRequest(userId, operation, units) { const creditsNeeded = this.calculateCreditsNeeded(operation, units); const userCredits = await this.getUserCredits(userId); if (userCredits < creditsNeeded) { return { success: false, message: 'Insufficient credits', needed: creditsNeeded, available: userCredits }; } await this.deductCredits(userId, creditsNeeded); return { success: true, creditsUsed: creditsNeeded }; } }

3. PAY-PER-USE MODEL

Perfect for high-value AI operations where users need occasional access to powerful capabilities.

// Usage-Based Pricing Calculator const usagePricing = { tiers: [ { min: 0, max: 100, pricePerUnit: 0.10 }, { min: 101, max: 1000, pricePerUnit: 0.08 }, { min: 1001, max: 10000, pricePerUnit: 0.05 }, { min: 10001, max: Infinity, pricePerUnit: 0.03 } ], calculateMonthlyBill(usage) { let bill = 0; let remaining = usage; for (let tier of this.tiers) { const tierUsage = Math.min(remaining, tier.max - tier.min + 1); bill += tierUsage * tier.pricePerUnit; remaining -= tierUsage; if (remaining <= 0) break; } // Add 3x markup for AI costs return bill * 3; } };

CHURN REDUCTION STRATEGIES

Reduce AI Tool Churn by 40%

  • Credit Rollover: Unused credits carry forward (max 2 months)
  • Pause Instead of Cancel: Offer subscription pause option
  • Win-Back Automation: Special offers for returning users
  • Usage Reminders: Alert users about unused credits
  • Feature Education: Weekly tips on new capabilities
  • Loyalty Rewards: Bonus credits for long-term subscribers
