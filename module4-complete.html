3. FREEMIUM WITH CREDITS

Users get free credits daily/monthly, can purchase more as needed.

// Credit system implementation
class CreditManager {
    constructor(userId) {
        this.userId = userId;
        this.freeCredits = 10; // Daily free credits
    }
    
    async checkCredits() {
        const user = await db.getUser(this.userId);
        const lastReset = user.lastCreditReset;
        const now = new Date();
        
        // Reset free credits daily
        if (now - lastReset > 86400000) {
            user.credits = this.freeCredits;
            user.lastCreditReset = now;
            await db.updateUser(user);
        }
        
        return user.credits;
    }
    
    async useCredit() {
        const credits = await this.checkCredits();
        if (credits > 0) {
            await db.decrementCredits(this.userId);
            return true;
        }
        return false; // Trigger purchase flow
    }
}

PRICE TESTING FRAMEWORK

A/B Testing Methodology

// Price testing implementation
class PriceTest {
    constructor() {
        this.variants = {
            A: { starter: 9, pro: 29, business: 99 },
            B: { starter: 7, pro: 27, business: 97 },
            C: { starter: 12, pro: 37, business: 127 }
        };
    }
    
    assignVariant(userId) {
        // Consistent assignment based on user ID
        const hash = this.hashCode(userId);
        const variantKeys = Object.keys(this.variants);
        const index = hash % variantKeys.length;
        return variantKeys[index];
    }
    
    getPricing(userId) {
        const variant = this.assignVariant(userId);
        this.trackView(userId, variant);
        return this.variants[variant];
    }
    
    trackConversion(userId, plan) {
        const variant = this.assignVariant(userId);
        analytics.track('Price Test Conversion', {
            userId,
            variant,
            plan,
            price: this.variants[variant][plan]
        });
    }
}

PAYMENT RECOVERY SYSTEM

// Dunning management for failed payments
class DunningManager {
    async handleFailedPayment(customerId) {
        const customer = await stripe.customers.retrieve(customerId);
        
        // Attempt 1: Immediate retry
        await this.retryPayment(customerId);
        
        // Attempt 2: After 3 days
        await this.scheduleRetry(customerId, 3);
        
        // Attempt 3: After 5 days with email
        await this.scheduleRetry(customerId, 5, {
            email: true,
            subject: 'Payment issue - action required',
            template: 'payment_failed_urgent'
        });
        
        // Attempt 4: After 7 days - last chance
        await this.scheduleRetry(customerId, 7, {
            email: true,
            subject: 'Final notice - account suspension tomorrow',
            template: 'final_notice',
            offer: '50% off next month if updated today'
        });
        
        // Day 10: Suspend account but keep data
        await this.suspendAccount(customerId);
    }
    
    // Recover 38% of failed payments
    async smartRetry(customerId) {
        const optimalTimes = [
            { day: 'Friday', hour: 9 },  // Payday
            { day: 'Monday', hour: 10 }, // Week start
            { day: 'Wednesday', hour: 14} // Mid-week
        ];
        
        for (let time of optimalTimes) {
            const result = await this.retryAtTime(customerId, time);
            if (result.success) break;
        }
    }
}

ADVANCED PRICING MODELS

Usage-Based Pricing

// Pay-per-use implementation
class UsageBasedBilling {
    constructor() {
        this.tiers = [
            { min: 0, max: 100, price: 0.10 },
            { min: 101, max: 1000, price: 0.08 },
            { min: 1001, max: 10000, price: 0.05 },
            { min: 10001, max: Infinity, price: 0.03 }
        ];
    }
    
    calculateMonthlyBill(usage) {
        let bill = 0;
        let remaining = usage;
        
        for (let tier of this.tiers) {
            const tierUsage = Math.min(
                remaining, 
                tier.max - tier.min + 1
            );
            bill += tierUsage * tier.price;
            remaining -= tierUsage;
            if (remaining <= 0) break;
        }
        
        return bill;
    }
}

Key Revenue Metrics

// Revenue analytics dashboard
const RevenueMetrics = {
    // Monthly Recurring Revenue
    MRR: async () => {
        const subs = await db.getActiveSubscriptions();
        return subs.reduce((sum, sub) => sum + sub.monthlyValue, 0);
    },
    
    // Average Revenue Per User
    ARPU: async () => {
        const revenue = await this.MRR();
        const users = await db.getActiveUserCount();
        return revenue / users;
    },
    
    // Customer Acquisition Cost
    CAC: async (period) => {
        const marketingSpend = await getMarketingSpend(period);
        const newCustomers = await getNewCustomers(period);
        return marketingSpend / newCustomers;
    },
    
    // Lifetime Value
    LTV: async () => {
        const arpu = await this.ARPU();
        const avgLifetime = await getAvgCustomerLifetime(); // in months
        return arpu * avgLifetime;
    },
    
    // LTV:CAC Ratio (should be > 3)
    LTVCACRatio: async () => {
        const ltv = await this.LTV();
        const cac = await this.CAC();
        return ltv / cac;
    }
};
